#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<bool> flag) {
    vector<int> X;
    for (int i = 0; i < arr.size(); i++) {
        if (flag[i]) {
            X.insert(X.end(), arr[i] * 2, arr[i]);
        } else {
            X.resize(X.size() - arr[i]);
        }
    }
    return X;
}
